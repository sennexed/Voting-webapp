version: '3.8'

services:
  mongodb:
    image: mongo:7.0
    container_name: gdc-mongodb
    ports:
      - "27017:27017"
    volumes:
      - mongodb_data:/data/db
    networks:
      - gdc-network

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: gdc-backend
    ports:
      - "8000:8000"
    environment:
      - MONGO_URL=mongodb://mongodb:27017
      - DATABASE_NAME=gdc_voting
    depends_on:
      - mongodb
    networks:
      - gdc-network

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    container_name: gdc-frontend
    ports:
      - "3000:80"
    depends_on:
      - backend
    networks:
      - gdc-network

volumes:
  mongodb_data:

networks:
  gdc-network:
    driver: bridge
